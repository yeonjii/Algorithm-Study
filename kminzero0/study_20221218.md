#### 백준 - 빙산
```python
from collections import deque

n, m = map(int, input().split())
graph = [list(map(int, input().split())) for _ in range(n)]
year = 0

dx = [-1, 1, 0, 0]
dy = [0, 0, -1, 1]

def bfs(a, b):
    queue = deque([[a, b]])
    visited[a][b] = True

    while queue:
        x, y = queue.popleft()

        for i in range(4):
            nx = x + dx[i]
            ny = y + dy[i]

            if nx < 0 or ny < 0 or nx >= n or ny >= m:
                continue
            
            if visited[nx][ny]:
                continue

            if graph[nx][ny] != 0:
                visited[nx][ny] = True
                queue.append([nx, ny])
            else:
                count[x][y] += 1
    return 1


while True:
    answer = []
    visited = [[False] * m for _ in range(n)]
    count = [[0] * m for _ in range(n)]

    for i in range(n):
        for j in range(m):
            if graph[i][j] != 0 and not visited[i][j]:
                answer.append(bfs(i, j))

    for i in range(n):
        for j in range(m):
            graph[i][j] -= count[i][j]
            if graph[i][j] < 0:
                graph[i][j] = 0

    if len(answer) == 0 or len(answer) >= 2:
        break

    year += 1

if len(answer) >= 2:
    print(year) 
else:
    print(0)
```
